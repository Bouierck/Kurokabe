def updateSensitive()
        if @page <=1 then
                @boutonRetour.set_sensitive(false)
                @boutonAvance.set_sensitive(true)
        elsif @page>=7 then
                @boutonAvance.set_sensitive(false)
                @boutonRetour.set_sensitive(true)
        else
                @boutonAvance.set_sensitive(true)
                @boutonRetour.set_sensitive(true)
        end
end
